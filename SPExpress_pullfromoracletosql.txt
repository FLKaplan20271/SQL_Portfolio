This is about the time when I worked at SP Express

I wrote this query so I could get data from one database into another. (Oracle to SQL Server)

SELECT
ORDERHDR.ORDERID, 
  ORDERHDR.SHIPID,   ORDERHDR.CUSTID, 
  MULTISHIPDTL.TRACKID,
  ORDERHDR.ORDERTYPE,   ORDERHDR.ENTRYDATE, 
  ORDERHDR.SHIPDATE,   ORDERHDR.PO, 
  ORDERHDR.RMA,   ORDERHDR.ORDERSTATUS, 
  ORDERHDR.COMMITSTATUS,   ORDERHDR.FROMFACILITY, 
  ORDERHDR.QTYORDER,   ORDERHDR.WEIGHTORDER, 
  ORDERHDR.CUBEORDER,   ORDERHDR.AMTORDER, 
  ORDERHDR.QTYCOMMIT,   ORDERHDR.WEIGHTCOMMIT, 
  ORDERHDR.CUBECOMMIT,   ORDERHDR.AMTCOMMIT, 
  ORDERHDR.QTYSHIP,   ORDERHDR.WEIGHTSHIP, 
  ORDERHDR.CUBESHIP,   ORDERHDR.AMTSHIP, 
  ORDERHDR.QTYTOTCOMMIT,   ORDERHDR.WEIGHTTOTCOMMIT, 
  ORDERHDR.CUBETOTCOMMIT,   ORDERHDR.AMTTOTCOMMIT, 
  ORDERHDR.QTYRCVD,   ORDERHDR.WEIGHTRCVD, 
  ORDERHDR.CUBERCVD,   ORDERHDR.AMTRCVD, 
  ORDERHDR.COMMENT1,   ORDERHDR.STATUSUSER, 
  ORDERHDR.STATUSUPDATE,   ORDERHDR.LASTUSER, 
  ORDERHDR.LASTUPDATE,   ORDERHDR.BILLOFLADING, 
  ORDERHDR.PRIORITY,   ORDERHDR.SHIPPER, 
  ORDERHDR.ARRIVALDATE,   ORDERHDR.CONSIGNEE, 
  ORDERHDR.SHIPTYPE,   ORDERHDR.CARRIER, 
  ORDERHDR."REFERENCE",   ORDERHDR.SHIPTERMS, 
  ORDERHDR.WAVE,   ORDERHDR.STAGELOC, 
  ORDERHDR.QTYPICK,   ORDERHDR.WEIGHTPICK, 
  ORDERHDR.CUBEPICK,   ORDERHDR.AMTPICK, 
  ORDERHDR.SHIPTONAME 
FROM ORDERHDR
JOIN MULTISHIPDTL ON ORDERHDR.SHIPID=MULTISHIPDTL.SHIPID and
ORDERHDR.ORDERID=MULTISHIPDTL.ORDERID
WHERE 
(ORDERHDR.DATESHIPPED BETWEEN TRUNC(CURRENT_DATE, 'IW') - 7 AND TRUNC(CURRENT_DATE, 'IW') - 1)
Order by ORDERHDR.CUSTID
